279 template <typename Sndr>
280 auto fire_and_forget(::std::span<::std::byte> storage,
281 Sndr sndr,
282 ::beman::execution::parallel_scheduler_replacement::receiver_proxy& proxy) const noexcept
283 -> void {
284 using inlined_holder = typename base::template inlined_state_holder<Sndr>;
285 using allocated_holder = typename base::template allocated_state_holder<Sndr, Alloc>;
286 constexpr bool inlined =
287 sizeof(inlined_holder) <= psched_storage_size && alignof(inlined_holder) <= psched_storage_alignment;
288 try {
289 if constexpr (inlined) {
290 auto holder = ::new (static_cast<void*>(storage.data())) inlined_holder(::std::move(sndr), proxy);
291 holder->do_start();
292 } else {
293 auto holder =
294 ::beman::execution::detail::new_object<allocated_holder>(alloc, ::std::move(sndr), proxy, alloc);
295 holder->do_start();
296 }
297 } catch (...) {
298 proxy.set_error(::std::current_exception());
299 }
300 }
301
302 [[no_unique_address]] Alloc alloc;
303};

307class task_scheduler {
308 public:
309 using scheduler_concept = ::beman::execution::scheduler_tag;
310 class sender;
311 struct domain {
312 struct not_a_scheduler {};
313 template <typename Sndr>
314 requires ::std::same_as<::beman::execution::tag_of_t<Sndr>, ::beman::execution::bulk_chunked_t> ||
315 ::std::same_as<::beman::execution::tag_of_t<Sndr>, ::beman::execution::bulk_unchunked_t>
316 static auto transform_sender(::beman::execution::set_value_t, Sndr&& sndr, const auto& env) noexcept {
317 auto&& [_, data, child] = ::std::forward<Sndr>(sndr);
318 auto [policy, shape, fn] = ::beman::execution::detail::forward_like<Sndr>(data);
319 auto sch = ::beman::execution::detail::call_with_default(
320 ::beman::execution::get_completion_scheduler<::beman::execution::set_value_t>,
321 not_a_scheduler{},
322 ::beman::execution::get_env(child),
323 ::beman::execution::detail::fwd_env(env));
324
325 static_assert(::std::same_as<decltype(sch), task_scheduler>,
326 "the completion scheduler of `sndr` shall be a `task_scheduler`.");
327
328 return ::beman::execution::detail::psched_bulk_sender{
329 ::std::bool_constant<
330 ::std::same_as<::beman::execution::tag_of_t<Sndr>, ::beman::execution::bulk_chunked_t>>{},
331 sch.backend_,
332 ::std::move(policy),
333 ::std::move(shape),
334 ::std::move(fn),
335 ::beman::execution::detail::forward_like<Sndr>(child)};
336 }
337 };

432 template <::beman::execution::receiver Rcvr>
433 class state {
434 public:
435 using operation_state_concept = ::beman::execution::operation_state_tag;
436 using stop_token_t = ::beman::execution::stop_token_of_t<::beman::execution::env_of_t<Rcvr>>;
437
438 state(::std::shared_ptr<backend_type> backend,
439 Rcvr&& rcvr) noexcept(::std::is_nothrow_constructible_v<::std::remove_cvref_t<Rcvr>, Rcvr>)
440 : backend_(::std::move(backend)), proxy_(::std::forward<Rcvr>(rcvr)) {}
441
442 auto start() & noexcept -> void {
443 auto stop_token =
444 proxy_.template try_query<::beman::execution::inplace_stop_token>(::beman::execution::get_stop_token);
445 if (stop_token.has_value() && stop_token->stop_requested()) {
446 proxy_.set_stopped();
447 return;
448 }
449 this->backend_->schedule(proxy_, ::std::span<::std::byte>{this->storage_});
450 }
451
452 private:
453 ::std::shared_ptr<backend_type> backend_;
454 rcvr_proxy<::std::remove_cvref_t<Rcvr>> proxy_;
455 alignas(::beman::execution::detail::psched_storage_alignment)::std::byte
456 storage_[::beman::execution::detail::psched_storage_size];
457 };

477 }
