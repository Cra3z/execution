37template <typename Sig>
38struct tuple_for_signature;

40template <typename Tag, typename... Args>
41struct tuple_for_signature<Tag(Args...)> {
43 using type = ::std::tuple<Tag, ::std::remove_cvref_t<Args>...>;
44
46 static auto select_tuple(Tag t, Args&&... args) -> type;
47};

50template <typename Signatures>
51struct variant_for_signatures;
52
53template <typename... Sigs>
54struct variant_for_signatures<::beman::execution::completion_signatures<Sigs...>>
55 : ::beman::execution::detail::tuple_for_signature<Sigs>... {
57 using type =
58 ::std::variant<std::monostate, typename ::beman::execution::detail::tuple_for_signature<Sigs>::type...>;
59
63 template <typename Tag, typename... Args>
64 static auto store_into_variant(type& storage, Tag t, Args&&... args) noexcept {
65 using tuple_type = decltype(select_tuple(t, ::std::forward<Args>(args)...));
66 try {
67 storage.template emplace<tuple_type>(t, ::std::forward<Args>(args)...);
68 } catch (...) {
69 if constexpr (!std::is_nothrow_constructible_v<tuple_type, Tag, Args&&...>) {
70 store_into_variant(storage, ::beman::execution::set_error_t{}, ::std::current_exception());
71 } else {
72 ::beman::execution::detail::unreachable();
73 }
74 }
75 }
76
80 template <::beman::execution::receiver Receiver>
81 static void complete_from_variant(type&& storage, Receiver&& receiver) noexcept {
82 std::visit(
83 [&](auto&& tuple_data) {
84 using tuple_type = ::std::remove_cvref_t<decltype(tuple_data)>;
85 try {
86 if constexpr (std::is_same_v<tuple_type, std::monostate>) {
87 ::beman::execution::detail::unreachable();
88 } else {
89 ::std::apply(
90 [&](const auto tag, auto&&... args) noexcept {
91 tag(::std::forward<Receiver>(receiver), ::std::forward<decltype(args)>(args)...);
92 },
93 // Make a copy, in case the tuple is destroyed by the completion call.
94 tuple_type{::std::move(tuple_data)});
95 }
96 } catch (...) {
97 if constexpr (!std::is_nothrow_constructible_v<tuple_type, tuple_type&&>) {
98 ::beman::execution::set_error(::std::forward<Receiver>(receiver), ::std::current_exception());
99 } else {
100 ::beman::execution::detail::unreachable();
101 }
102 }
103 },
104 ::std::move(storage));
105 }
106
108 using ::beman::execution::detail::tuple_for_signature<Sigs>::select_tuple...;
109};

112template <typename Signatures>
113struct completion_storage {
114 completion_storage() noexcept = default;
115
117 template <typename Tag, typename... Args>
118 void store(Tag t, Args&&... args) noexcept {
119 variant_impl::store_into_variant(storage, t, ::std::forward<Args>(args)...);
120 }
121
123 template <::beman::execution::receiver_of<Signatures> Receiver>
124 void complete(Receiver&& receiver) && noexcept {
125 variant_impl::complete_from_variant(::std::move(storage), ::std::forward<Receiver>(receiver));
126 }
127
128 private:
129 using variant_impl = ::beman::execution::detail::variant_for_signatures<Signatures>;
130 typename variant_impl::type storage;
131};
