89 template <::beman::execution::enter_scope_sender Scope, ::beman::execution::sender Work>
90 auto operator()(Scope&& scope, Work&& work) const
91 noexcept(::std::is_nothrow_constructible_v<::std::remove_cvref_t<Scope>, Scope> &&
92 ::std::is_nothrow_constructible_v<::std::remove_cvref_t<Work>, Work>) {
93 return ::beman::execution::detail::make_sender(
94 *this,
95 ::beman::execution::detail::product_type{::std::forward<decltype(scope)>(scope),
96 ::std::forward<decltype(work)>(work)});

224template <::beman::execution::enter_scope_sender Scope,
225 ::beman::execution::sender Work,
226 ::beman::execution::receiver Receiver>
227struct within_t::state {
228 using operation_state_concept = ::beman::execution::operation_state_tag;
229 using self_t = state<Scope, Work, Receiver>;
230 using scope_t = ::std::remove_cvref_t<Scope>;
231 using work_t = ::std::remove_cvref_t<Work>;
232 using env_t = ::beman::execution::env_of_t<Receiver>;
233 using exit_scope_sender_t = ::beman::execution::exit_scope_sender_of_t<scope_t, env_t>;
234 using enter_receiver_t = ::beman::execution::detail::within_enter_receiver<self_t, env_t>;
235 using work_receiver_t = ::beman::execution::detail::within_work_receiver<self_t, env_t>;
236 using exit_receiver_t = ::beman::execution::detail::within_exit_receiver<self_t, env_t>;
237 using enter_op_t = ::beman::execution::connect_result_t<scope_t, enter_receiver_t>;
238 using work_op_t = ::beman::execution::connect_result_t<work_t, work_receiver_t>;
239 using exit_op_t = ::beman::execution::connect_result_t<exit_scope_sender_t, exit_receiver_t>;
240 using work_completions_t = ::beman::execution::completion_signatures_of_t<work_t, env_t>;
241
243 struct enter_state {
245 enter_op_t op;
247 work_t work;
248 };
249
251 struct work_state {
253 work_op_t op;
255 exit_scope_sender_t exit_sender;
256 };
257
259 struct exit_state {
261 exit_op_t op;
263 ::beman::execution::detail::completion_storage<work_completions_t> work_completion;
264 };
265
267 ::std::remove_cvref_t<Receiver> receiver;
269 std::variant<enter_state, work_state, exit_state> state_;
270
272 template <::beman::execution::sender S, ::beman::execution::sender W>
273 state(S&& scope, W&& work, Receiver&& r)
274 : receiver(std::forward<Receiver>(r)),
275 state_(std::in_place_type<enter_state>, ::beman::execution::detail::elide([&]() noexcept {
276 return enter_state{
277 ::beman::execution::connect(::std::forward<S>(scope), enter_receiver_t{{*this}}),
278 ::std::forward<W>(work)};
279 })) {}
280
282 auto start() noexcept -> void {
283 // Start the operation for the enter scope sender; this would eventually trigger the rest of the work.
284 const auto p = std::get_if<enter_state>(&state_);
285 assert(p);
286 ::beman::execution::start(p->op);
287 }
288
290 auto get_env() const noexcept -> env_t { return ::beman::execution::get_env(this->receiver); }
291
293 void on_enter_set_value(exit_scope_sender_t&& exit_sender) noexcept {
294 constexpr auto nothrow =
295 ::beman::execution::detail::nothrow_callable<::beman::execution::connect_t, work_t, work_receiver_t>;
296
297 const auto p = std::get_if<enter_state>(&state_);
298 assert(p);
299 try {
300 auto work_sender = ::std::move(p->work);
301 auto& new_state =
302 this->state_.template emplace<work_state>(::beman::execution::detail::elide([&]() noexcept {
303 return work_state{::beman::execution::connect(::std::move(work_sender), work_receiver_t{{*this}}),
304 ::std::move(exit_sender)};
305 }));
306 ::beman::execution::start(new_state.op);
307 } catch (...) {
308 if constexpr (nothrow) {
309 ::beman::execution::detail::unreachable();
310 } else {
311 ::beman::execution::set_error(::std::move(this->receiver), ::std::current_exception());
312 }
313 }
314 }
315
317 template <typename E>
318 void on_enter_set_error(E&& e) noexcept {
319 ::beman::execution::set_error(::std::move(this->receiver), ::std::forward<E>(e));
320 }
321
323 void on_enter_set_stopped() noexcept { ::beman::execution::set_stopped(::std::move(this->receiver)); }
324
326 template <typename Tag, typename... Args>
327 void on_work_completion(Tag t, Args&&... args) noexcept {
328 const auto p = std::get_if<work_state>(&state_);
329 assert(p);
330
331 // Put the args in the completion storage, before destroying the operation state.
332 ::beman::execution::detail::completion_storage<work_completions_t> completion;
333 completion.store(t, ::std::forward<Args>(args)...);
334 auto exit_sender = ::std::move(p->exit_sender);
335 // Replace the state as we transition to "exit" state.
336 auto& new_state = this->state_.template emplace<exit_state>(::beman::execution::detail::elide([&]() noexcept {
337 return exit_state{::beman::execution::connect(::std::move(exit_sender), exit_receiver_t{{*this}}),
338 std::move(completion)};
339 }));
340 ::beman::execution::start(new_state.op);
341 }
342
344 void on_exit_set_value() noexcept {
345 const auto p = std::get_if<exit_state>(&state_);
346 assert(p);
347 std::move(p->work_completion).complete(::std::move(this->receiver));
348 }
349};
